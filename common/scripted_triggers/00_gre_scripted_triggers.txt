    for_each_loop = {         
        array = enemy_id
        
        if = { 
            limit = { check_variable = { v = THIS.id } }
            set_temp_variable = { x = i }
            break = 1
        } 
    }
    for_each_loop = {         
        array = enemy_as
        
        if = { 
            limit = { check_variable = { x = i } }

        } 
    }

}
