pursuing_the_balkan_pact = {                                        ## Greece pursues the Balkan pact by completing the focus.
    allowed = {
        OR = {
            original_tag = YUG
            original_tag = ROM
            original_tag = TUR
            original_tag = BUL
            original_tag = ALB
        }
    }
    enable = {
        GRE = { has_completed_focus = GRE_the_balkan_pact }
        if = { limit = { tag = BUL } BUL = { has_country_flag = joined_the_balkan_pact } }
        if = { limit = { tag = ALB } ALB = { has_country_flag = joined_the_balkan_pact } }
    }
    abort_when_not_enabled = yes
    
    ai_strategy = {
        type = befriend
        id = GRE
        value = 5
    }
}
