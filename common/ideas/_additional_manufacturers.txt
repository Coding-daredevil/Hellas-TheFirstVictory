			picture = gre_exiled_armor_manufacturer

			allowed = {
				original_tag = GRE
				has_dlc = "Battle for the Bosporus"
			}
            
            visible = {
                tag = GRE
                has_axis_occupation = yes
            }
            
            removal_cost = 75
            
            on_add = {
                add_to_variable = { GRE.current_manufacturers = 1 }
                hidden_effect = {
                    if = {
                        limit = { check_variable = { var = GRE.current_manufacturers value = GRE.max_manufacturers compare = greater_than_or_equals  } }
                        if = { 
                            limit = { NOT = { has_idea = GRE_exiled_naval_manufacturer } }
                            add_ideas = GRE_exiled_naval_timeout

                subtract_from_variable = { GRE.current_manufacturers = 1 }
                if = {
                    limit = { check_variable = { var = GRE.current_manufacturers value = GRE.max_manufacturers compare = less_than  } }

                        limit = { has_idea = GRE_exiled_air_timeout }
                        remove_ideas = GRE_exiled_air_timeout

                    if = {
                        limit = { has_idea = GRE_exiled_armor_timeout }
